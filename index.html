Get Started in Minutes

Install the SDK and deploy your first backend.

// Install BASESTACK SDK
npm install @basestack/sdk

// Initialize your app
import { Basestack } from '@basestack/sdk';

const app = new Basestack({
  apiKey: 'your-api-key'
});

// Deploy a simple endpoint
app.post('/api/hello', (req, res) => {
  res.json({ message: 'Hello from BASESTACK!' });
});

app.listen();
